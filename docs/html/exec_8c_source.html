150 int exec_should_stop_at_floor(floor_codes_t current_floor) {
151 
152  inside_queue_t *inside_queue_ptr = order_get_inside_queue();
153  outside_queue_t *outside_queue_ptr = order_get_outside_queue();
154 
155  if (current_floor == destination_floor) {
156  return 1;
157  }
158  // Check inside orders
159  else if (inside_queue_ptr->length) {
160  for(int i = 0; i < inside_queue_ptr->length; i++) {
161  if(inside_queue_ptr->queue[i].floor == current_floor)
162  {
163  return 1;
164  }
165  }
166  }
167  // Check outside orders
168  else if (outside_queue_ptr->length) {
169  for(int i = 0; i < outside_queue_ptr->length; i++)
170  {
171  if (current_floor == floor_1 || current_floor == floor_4) {
172  if(outside_queue_ptr->queue[i].floor == current_floor)
173  {
174  return 1;
175  }
176  }
177  else if (outside_queue_ptr->queue[i].floor == current_floor && last_direction == DIRN_STOP) {
178  return 1;
179  }
180  else if(outside_queue_ptr->queue[i].floor == current_floor && outside_queue_ptr->queue[i].direction == last_direction)
181  {
182  return 1;
183  }
184  }
185  }
186  //If nothing found, return 0
187  return 0;
188 }
